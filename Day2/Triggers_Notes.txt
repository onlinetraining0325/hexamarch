Create table Student
(
    Sno number primary key,
    name varchar2(30),
    sub1 number,
    sub2 number,
    sub3 number,
    total number,
    aveg number(9,2)
);

Create or replace trigger trgStudentUpdate AFTER Insert ON Student
BEGIN
    Update Student set total = (sub1+sub2+sub3), aveg = (sub1+sub2+sub3)/3;
END;

There are various types of triggers we have as 

BEFORE Trigger : This trigger executes before the INSERT 

AFTER Trigger

Instead Of Trigger 

Assume, i want to insert record into Employ table, but empno not > 100 that condition we have to check by using Trigger 

A trigger contains 2 keywords as :new, :old

:new contains all the recently inserted record value 

:old contains deleted/updated value records 

CREATE OR REPLACE TRIGGER trgEmpOperation
BEFORE INSERT ON EMPLOY
FOR EACH ROW
BEGIN
    IF :NEW.empno >= 100 THEN
        RAISE_APPLICATION_ERROR(-20001, 
            'Employee No cannot be >= 100');
    END IF;
END;

Create Table Customer
(
   Id INT Primary Key,
   Name varchar2(30),
   City varchar2(30),
   Amount number
);


Create Table CustomerBackup
(
   Id INT,
   Name varchar2(30),
   City varchar2(30),
   Amount number,
   Operation varchar2(30),  
   OperationDate Date default sysdate
);

-- Create a trigger ensures all inserted records to be stored in CustomerBackup table 
with Operation Name 

create or replace trigger trgCustomerInsert AFTER INSERT ON Customer 
FOR EACH ROW
BEGIN
    Insert into CustomerBackup(id,name,city,amount,operation)
	values(:new.id,:new.name,:new.city,:new.amount,'Insert Operation');
END; 

-- Create a trigger ensures all old records of updated to be stored in CustomerBackup table 

Create or replace trigger trgCustomerUpdate AFTER UPDATE ON CUSTOMER
FOR EACH ROW
BEGIN
     Insert into CustomerBackup(id,name,city,amount,operation)
	values(:old.id,:old.name,:old.city,:old.amount,'Update Operation');
END;

Update Customer set name='Lingeshwar',City='Chennai',amount=88823 where id=1;

-- Create a trigger ensures all deleted records to be stored in CustomerBackup table

create or replace trigger trgCustomerDelete AFTER DELETE ON CUSTOMER
FOR EACH ROW
BEGIN
 Insert into CustomerBackup(id,name,city,amount,operation)
	values(:old.id,:old.name,:old.city,:old.amount,'Delete Operation');
END;

Instead of trigger to be created on Views 

Create Table EmpNew
(
   Eid Number primary key,
   nam varchar(30),
   sal numeric(9,2)
);


Create Table LeaveHistory
(
   leaveId number primary key,
   eid number references EmpNew(eid),
   leaveReason varchar2(30) 
);

create view vwEmpLeaves
as
   select E1.Eid,E1.nam,E1.sal,Lh.leaveId,Lh.leaveReason
     from EmpNew E1 INNER JOIN LeaveHistory LH on
	E1.Eid=Lh.Eid;


Create or replace trigger trg_View_Ins
INSTEAD OF INSERT ON vwEmpLeaves
FOR EACH ROW
BEGIN
    Insert into EmpNew(Eid,nam,sal) values(:new.eid,:new.nam,:new.sal);
    Insert into LeaveHistory(leaveId,eid,leaveReason) values(:new.leaveId,
	:new.eid,:new.leaveReason);
END;
